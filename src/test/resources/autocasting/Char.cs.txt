namespace autocasting
{
	internal class CharAutocasting
	{
		private char _ch;

		private char test()
		{
			this._ch = (char)1;
			switch (this._ch)
			{
				case (char)0x0022:
				{
					return (char)0x0033;
				}
			}
			return ' ';
		}

		public virtual char ch(int index)
		{
			string str = "teststr";
			return index < str.Length ? str[index] : (char)0x0000;
		}
	}
}
