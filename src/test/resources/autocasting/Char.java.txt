package autocasting;

class CharAutocasting {
	private char _ch;
	
	private char test() {
		this._ch = 1;

		switch (this._ch)
        {
            case 0x0022:
                return 0x0033;
        }

        return ' ';
	}

	public char ch(int index)
    {
        String str = "teststr";
        return index < str.length() ? str.charAt(index) : 0x0000;
    }
}
